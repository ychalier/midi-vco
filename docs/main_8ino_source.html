110 void update_midi_led()
111 {
112  if (current_midi_led_state && (millis() - midi_led_high_time_ms > MIDI_LED_TIMEOUT))
113  {
114  current_midi_led_state = false;

116  }
117  previous_midi_led_state = current_midi_led_state;
